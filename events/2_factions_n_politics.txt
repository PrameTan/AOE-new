country_event = {

		days = 1
	}
	
	option = {
		name = "Factions SetUp II"
		#Church
		any_country = {

					continent = asia
				}

			}

				add_province_modifier = {
					name = asian_farms
					duration = -1
				}
			}
		}
	}
}
