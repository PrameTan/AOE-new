country_event = {
	title = "Colony Crumbles"
	desc = "Colony Crumbles."
	#picture = "Education"
	#major = yes

	id = 9999816

	trigger = {
	           has_country_flag = dynamic_tag
			   AND = {
			         is_vassal = no
			         is_substate = no
			   }
			   exists = yes
	}
	
	mean_time_to_happen = {
		days = 1
	}
    option = { 
	          name = "Ok" 
			  all_core = { remove_core = THIS }
		      any_country = {
		      	             limit = { vassal_of = THIS }
		      	             leave_alliance = THIS
		      	             country_event = 31523
		      }
		      any_country = {
		      	limit = {
		      		neighbour = THIS
		      		exists = yes
		      		any_core = { owned_by = THIS }
		      	}
		      	country_event = 9999814
		      }
		      any_country = {
		      	limit = {
		      		any_core = { owned_by = THIS }
		      		exists = no
		      		is_cultural_union = no
		      	}
		      	country_event = 9999815
		      }
	}
}
